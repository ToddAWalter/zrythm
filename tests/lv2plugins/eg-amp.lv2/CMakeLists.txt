  manifest.ttl.in
  ${CMAKE_CURRENT_BINARY_DIR}/manifest.ttl
  @ONLY)

  amp.ttl
  ${CMAKE_CURRENT_BINARY_DIR}/amp.ttl
  COPYONLY)

  amp.c)

  OUTPUT_NAME "amp")

  zrythm::lv2)

  "eg_amp|http://lv2plug.in/plugins/eg-amp|${CMAKE_CURRENT_BINARY_DIR}|eg_amp_lv2")
