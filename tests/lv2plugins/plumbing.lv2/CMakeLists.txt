  manifest.ttl
  ${CMAKE_CURRENT_BINARY_DIR}/manifest.ttl

  plumbing.ttl
  ${CMAKE_CURRENT_BINARY_DIR}/plumbing.ttl

  "plumbing|http://lv2plug.in/plugins/plumbing|${CMAKE_CURRENT_BINARY_DIR}|plumbing_lv2")
